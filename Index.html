#particle-canvas { position: absolute; top: 0; left: 0; width: 100%; height: 100%; background-color: #000000; } const canvas = document.getElementById('particle-canvas'); const ctx = canvas.getContext('2d'); const particles = []; function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); for (let i = 0; i < particles.length; i++) { const p = particles[i]; ctx.beginPath(); ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2); ctx.fillStyle = p.color; ctx.fill(); p.x += p.vx; p.y += p.vy; } requestAnimationFrame(draw); } draw();
